常用功能

定时器

  1. 开启定时器
    function showAlert() {
        alert("22")
    }
    //每过3s执行一次（3s后执行第一次）
    setInterval(showAlert, 3000)
    //1s后只执行1次
    setTimeout(showAlert, 1000)
  1. 关闭定时器
var timer =  setInterval(showAlert, 3000)
clearInterval(timer)
var timer =  setTimeout(showAlert, 1000)
clearTimeout(timer)

日期

var date = new Date()

date.getFullYear()
date.getDay() //获取星期
date.getDate() //获取日
date.getHours()

大小写

str.toUpperCase()

"avcdfg".search("g")
